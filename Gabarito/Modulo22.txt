DROP VIEW V_JUNCAO01;
CREATE VIEW V_JUNCAO01 AS
SELECT 
F.*, T.Ano,T.Mes,T.Dia,P.IDADE, P.FAIXA_ETARIA, P.GENERO, M.MUNICIPIO_RESIDENCIA,M.POPULAÇÃO_ESTIMADA POPULACAO, S.INTERNACAO,S.INTERNACAO, MACRO, MICRO, M.PIB_PER_CAPITA
FROM Fato_Internacao F
INNER JOIN DIM_TEMPO T ON T.ID_TEMPO=F.ID_TEMPO
INNER JOIN DIM_PACIENTE P ON P.ID_PESSOA=F.ID_PESSOA
INNER JOIN DIM_MUNICIPIO M ON M.ID_MUNICIPIO=F.ID_MUNICIPIO
INNER JOIN DIM_STATUS_TRATAMENTO S ON S.ID_PESSOA=F.ID_PESSOA;

/*(1) Encontre a frequências de casos por gênero e grupo etário. Faça uma avaliação visual e identifique o grupo etário modal 
de ambas as distribuições, Feminino e Masculino.  */
SELECT  
GENERO
,FAIXA_ETARIA
, COUNT(1) AS TOTAL 
FROM V_JUNCAO01
GROUP BY GENERO
,FAIXA_ETARIA
ORDER BY GENERO,FAIXA_ETARIA;

--(2) Encontre a idade média de casos de COVID por gênero. 
SELECT  
GENERO
,avg(idade)
FROM V_JUNCAO01
GROUP BY GENERO
ORDER BY GENERO;

/*3) Encontre a taxa de pessoas que pegaram COVID no período por macro região, classifique nas ordem descendente por taxa.
 (Taxa = 100.000 x número casos / População). Identifique as cinco macro regiões com pior situação, de acordo com o valor da taxa. 
 */
SELECT  
MACRO
, ROUND(100000.0*COUNT(1)/(SELECT SUM (POPULACAO) FROM DIM_MUNICIPIO M WHERE M.MACRO=V.MACRO),2) AS TAXA 
FROM V_JUNCAO01 V
GROUP BY MACRO
ORDER BY TAXA desc;

/*
(4) Encontre o valor médio do PIB per capita médio de cada macro de regiões para o resultado da consulta da pergunta (3) .  
(valor médio do PIB per capta da macro região = PIB_PER_CAPTA * POPULACAO  do Município 1 + PIB_PER_CAPTA * POPULACAO  do Município 2 + ... + PIB_PER_CAPTA * POPULACAO  do Município N) / (POPULACAO do município 1 + ...+ POPULACAO do município N). 
*/
SELECT  
MACRO
, ROUND(SUM (PIB_PER_CAPITA*POPULAÇÃO_ESTIMADA)/SUM(POPULAÇÃO_ESTIMADA),2) AS PIB_PER_CAPITA_MACRO
FROM DIM_MUNICIPIO V
GROUP BY MACRO
ORDER BY PIB_PER_CAPITA_MACRO desc;


/*
(5) Concatene o resultado da consulta (4) com o resultado da consulta(3).
*/
SELECT  
PIB_PER_CAPITA_MACRO, A1.*
FROM (SELECT  
MACRO
, ROUND(100000.0*COUNT(1)/(SELECT SUM (POPULACAO) FROM DIM_MUNICIPIO M WHERE M.MACRO=V.MACRO),2) AS TAXA 
FROM V_JUNCAO01 V
GROUP BY MACRO
ORDER BY TAXA desc) A1
INNER JOIN
(SELECT  
MACRO
, ROUND(SUM (PIB_PER_CAPITA*POPULAÇÃO_ESTIMADA)/SUM(POPULAÇÃO_ESTIMADA),2) AS PIB_PER_CAPITA_MACRO
FROM DIM_MUNICIPIO V
GROUP BY MACRO
ORDER BY PIB_PER_CAPITA_MACRO desc) A2
ON A1.MACRO=A2.MACRO
ORDER BY TAXA DESC;


/*
(6) Use a macro região com maior taxa e obtenha a frequência de casos de COVID mensalmente durante o período de 2020 e 2021. 
Identifique a partir de qual mês podemos dizer que os casos atingiram um pico e depois começaram a apresentar um comportamento de oscilação e veio diminuindo. 
*/
SELECT  ANO, MES, COUNT(1) AS TOTAL
FROM V_JUNCAO01 WHERE MACRO='JEQUITINHONHA'
GROUP BY ANO, MES;



/*
(7) Obtenha a distribuição percentual de casos, segundo cada mês e ano, usando para isso resultado da consulta da pergunta (6). 
*/
SELECT  ANO, MES, COUNT(1) AS TOTAL
FROM V_JUNCAO01 WHERE MACRO='JEQUITINHONHA'
GROUP BY ANO, MES;
