SELECT GRUPO_CURSO,AVG(NT_GER) 
FROM FATO_PROVA P
INNER JOIN DIM_CURSO C ON C.ID_CURSO=P.ID_CURSO
GROUP BY GRUPO_CURSO

SELECT * FROM (
SELECT GRUPO_CURSO,MEDIA,
DENSE_RANK() OVER (ORDER BY MEDIA DESC) AS RANQUE
 FROM (
SELECT GRUPO_CURSO,AVG(NT_GER) MEDIA
FROM FATO_PROVA P
INNER JOIN DIM_CURSO C ON C.ID_CURSO=P.ID_CURSO
GROUP BY GRUPO_CURSO
ORDER BY MEDIA) ) WHERE RANQUE<6

SELECT GRUPO_CURSO,tp_sexo,RAÇA,AVG(NT_GER) MEDIA
FROM FATO_PROVA P
INNER JOIN DIM_CURSO C ON C.ID_CURSO=P.ID_CURSO
INNER JOIN DIM_INFO_ALUNO I ON I.ID_ALUNO_INFO=P.ID_ALUNO_INFO
WHERE GRUPO_CURSO='MEDICINA'
GROUP BY GRUPO_CURSO,tp_sexo,RAÇA
ORDER BY MEDIA;


SELECT NOME_IES,count(NT_GER) AS NOTA
FROM FATO_PROVA P
INNER JOIN DIM_IES I ON I.ID_IES_ANO=P.ID_IES_ANO 
WHERE P.ID_TEMPO=2019
GROUP BY NOME_IES
ORDER BY NOTA desc;

select COUNT(1) from DIM_IES i
where i.ID_IES_ANO NOT in (select p.ID_IES_ANO from FATO_PROVA p where ID_TEMPO=2019)
